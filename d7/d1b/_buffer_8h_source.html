159 template<class T> void Buffer<T>::add(const T& data, bool dropIfFull)
160 {
161  // Acquire semaphore
162  clearBuffer_add->acquire();
163 
164  // If dropping is enabled, do not block if buffer is full
165  if(dropIfFull)
166  {
167  // Try and acquire semaphore to add item
168  if(freeSlots->tryAcquire())
169  {
170  // Add item to queue
171  queueProtect.lock();
172  queue.enqueue(data);
173  queueProtect.unlock();
174 
175  // Release semaphore
176  usedSlots->release();
177  }
178  }
179  // If buffer is full, wait on semaphore
180  else
181  {
182  // Acquire semaphore
183  freeSlots->acquire();
184 
185  // Add item to queue
186  queueProtect.lock();
187  queue.enqueue(data);
188  queueProtect.unlock();
189 
190  // Release semaphore
191  usedSlots->release();
192  }
193  // Release semaphore
194  clearBuffer_add->release();
195 }

197 template<class T> T Buffer<T>::get()
198 {
199  // Local variable(s)
200  T data;
201 
202  // Acquire semaphores
203  clearBuffer_get->acquire();
204  usedSlots->acquire();
205 
206  // Take item from queue
207  queueProtect.lock();
208  data=queue.dequeue();
209  queueProtect.unlock();
210 
211  // Release semaphores
212  freeSlots->release();
213  clearBuffer_get->release();
214 
215  // Return item to caller
216  return data;
217 }

219 template<class T> bool Buffer<T>::clear()
220 {
221  // Check if buffer contains items
222  if(queue.size()>0)
223  {
224  // Stop adding items to buffer (will return false if an item is currently being added to the buffer)
225  if(clearBuffer_add->tryAcquire())
226  {
227  // Stop taking items from buffer (will return false if an item is currently being taken from the buffer)
228  if(clearBuffer_get->tryAcquire())
229  {
230  // Release all remaining slots in queue
231  freeSlots->release(queue.size());
232 
233  // Acquire all queue slots
234  freeSlots->acquire(bufferSize);
235 
236  // Reset usedSlots to zero
237  usedSlots->acquire(queue.size());
238 
239  // Clear buffer
240  queue.clear();
241 
242  // Release all slots
243  freeSlots->release(bufferSize);
244 
245  // Allow get method to resume
246  clearBuffer_get->release();
247  }
248  else
249  {
250  return false;
251  }
252 
253  // Allow add method to resume
254  clearBuffer_add->release();
255  return true;
256  }
257  else
258  {
259  return false;
260  }
261  }
262  else
263  {
264  return false;
265  }
266 }
