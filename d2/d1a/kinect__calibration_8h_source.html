36  //disparity2point: converts a point in the depth image to a 3D point
37  //in color camera coordinates.
38  void disparity2point(int u,int v,short disp,cv::Matx31f &pc) {
39  cv::Matx31f pd;
40 
41  pd(2) = 1.0f / (calib_dc1*(disp - calib_dc2));
42  pd(0) = ((u-calib_px_d) / calib_fx_d) * pd(2);
43  pd(1) = ((v-calib_py_d) / calib_fy_d) * pd(2);
44 
45  pc = calib_R*pd+calib_T;
46  }

48  //point2rgb: projects a 3D point in color camera coordinates onto the
49  //image plane of the color camera and returns the pixel coordinates.
50  void point2rgb(cv::Matx31f &pc, float &uc, float &vc)
51  {
52  uc = pc(0)*calib_fx_rgb/pc(2) + calib_px_rgb;
53  vc = pc(1)*calib_fy_rgb/pc(2) + calib_py_rgb;
54  }
